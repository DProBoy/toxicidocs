from TwitterCommon.utils import get_config
from TwitterCommon.errors import TooManyAPICalls
from googleapiclient import discovery
import json
import pandas as pd

# Allows the usage of the Perspective API from google to analyze toxicity of tweets
class Perspective:
    def __init__(self):
        self.connect()
        self.request_counter = 0
        pass

    def connect(self):
        API_KEY = get_config().get("perspective_api", "API_KEY")
        self.client = discovery.build(
            "commentanalyzer",
            "v1alpha1",
            developerKey=API_KEY,
            discoveryServiceUrl="https://commentanalyzer.googleapis.com/$discovery/rest?version=v1alpha1",
            static_discovery=False,
        )

    # Reference for attributes definition : https://developers.perspectiveapi.com/s/about-the-api-attributes-and-languages?language=en_US
    ALL_ATTRIBUTES = ['TOXICITY', 'SEVERE_TOXICITY',
                      'IDENTITY_ATTACK', 'INSULT', 'PROFANITY', 'THREAT']

    # New attributes, they're experimental
    ALL_EXPERIMENTAL_ATTRIBUTES = [attr + "_EXPERIMENTAL" for attr in ['TOXICITY', 'SEVERE_TOXICITY',
                                                                       'IDENTITY_ATTACK', 'INSULT', 'PROFANITY', 'THREAT']] + ['SEXUALLY_EXPLICIT', 'FLIRTATION']

    # Supports only ENGLISH
    NEWYORK_TIMES_ATTR = ['ATTACK_ON_AUTHOR', 'ATTACK_ON_COMMENTER', 'INCOHERENT',
                          'INFLAMMATORY', 'LIKELY_TO_REJECT', 'OBSCENE', 'SPAM', 'UNSUBSTANTIAL']

    def analyze_text(self, text, attributes):
        self.request_counter += 1
        if get_config().getint("perspective_api", "max_requests_per_instance") < self.request_counter:
            raise TooManyAPICalls()
        requestedAttributes = {}
        for attr in attributes:
            requestedAttributes[attr] = {}
        analyze_request = {
            'comment': {'text': text},
            'requestedAttributes': requestedAttributes
        }

        response = self.client.comments().analyze(body=analyze_request).execute()

        return response

    def perspective_analysis(self, attributes):
        def decorator(df):
            text = df['text']
            toxicity_df = pd.DataFrame(
                columns=attributes+["detected_language"])
            for i, v in text.iteritems():
                res = self.analyze_text(v, attributes)
                toxicity_dict = {attr: res["attributeScores"][attr]
                                 ["summaryScore"]["value"] for attr in attributes}
                toxicity_dict["detected_language"] = res["detectedLanguages"][0]
                toxicity_df = toxicity_df.append(
                    toxicity_dict, ignore_index=True)
            df.reset_index(drop=True, inplace=True)
            conc_res = pd.concat([df, toxicity_df], axis=1)
            # TODO : Fix extra unnamed cols
            return conc_res
        return decorator
    
    def get_score_from_text(self,text,nb):
        if nb == 0:
            attr = Perspective.ALL_ATTRIBUTES
        elif nb == 1:
            attr = Perspective.ALL_EXPERIMENTAL_ATTRIBUTES
        elif nb == 2:
            attr = Perspective.NEWYORK_TIMES_ATTR
        else:
            print("No valid attribue, please choose between 0,1 and 2")
            return None 
        
        myDetector = Perspective()
        data_json =json.loads(json.dumps(myDetector.analyze_text(text,attr)["attributeScores"]))
        dico = {"text": text}
        for key in data_json.keys():
            dico[key] = data_json[key]["summaryScore"]["value"]
        return pd.DataFrame(dico, index=[0])
    
    def get_toxicity(self,text):
        return (Perspective.get_score_from_text(self,text,1).iloc[0]["TOXICITY_EXPERIMENTAL"],Perspective.get_score_from_text(self,text,1).iloc[0]["SEVERE_TOXICITY_EXPERIMENTAL"])
