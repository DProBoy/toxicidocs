def static_vars(**kwargs):
    """
    Allows you to store static variables inside a function

    Usage Examples:
        @static_vars(count=0)
        def my_func():
            print(my_func.count)
            my_func.count += 1

        my_func()
        my_func()
        my_func()

        >>>> console:
        0
        1
        2

    Args:
        **kwargs: list of static variables to add to function

    """
    def decorate(func):
        for k in kwargs:
            setattr(func, k, kwargs[k])
        return func
    return decorate


def add_config(func):
    """
    Allows you to add a config to the function
    Usage Examples:
        @add_config
        def my_func():
            print(my_func.config)


        >>>> console:
        ...config
    """
    if not hasattr(func, "config"):
        # it doesn't exist yet, so initialize it
        func.config = configparser.ConfigParser()
        with open("./config.ini") as config_handle:
            func.config.read_file(config_handle)
    return func


@add_config
def get_config():
    """
        Returns a configparser of our config file
    """
    return get_config.config


@add_config
def mon(api):
    """
        LEGACY : (Deprecated) Tries to count the number of single API calls.
    """
    if mon.config.get("development", "quota_limiter"):
        counter = getattr(mon, 'counter', 0) + 1
        if counter > mon.config.getint("development", "quota_limit_per_session"):
            raise TooManyAPICalls()
            return None
    return api
